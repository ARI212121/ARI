function drawPointsOnMap(locations) { console.log("מתחיל עיבוד " + locations.length + " נקודות..."); let wkts = []; let names = []; let tooltips = []; let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity; let validPointsCount = 0; // 1. עיבוד הקואורדינטות for (let i = 0; i < locations.length; i++) { const loc = locations[i]; if (loc.lat && loc.lng) { try { const itmCoords = proj4("EPSG:4326", "EPSG:2039", [parseFloat(loc.lng), parseFloat(loc.lat)]); const x = Math.round(itmCoords[0]); const y = Math.round(itmCoords[1]); if (x > 130000 && x < 280000 && y > 380000 && y < 800000) { if(x < minX) minX = x; if(x > maxX) maxX = x; if(y < minY) minY = y; if(y > maxY) maxY = y; wkts.push("POINT(" + x + " " + y + ")"); names.push("p" + i); tooltips.push(loc.address || "נקודת ציון"); validPointsCount++; } } catch (e) { console.error("שגיאה בהמרת נקודה:", e); } } } if (validPointsCount > 0) { setTimeout(function() { // 2. קודם כל מבצעים זום ומציגים את הכפתורים (כדי להבטיח שזה יעבוד תמיד!) try { const marginX = (maxX - minX) * 0.05; const marginY = (maxY - minY) * 0.05; govmap.zoomToExtent({ xmin: minX - marginX, ymin: minY - marginY, xmax: maxX + marginX, ymax: maxY + marginY }); document.getElementById('loading').style.display = 'none'; document.getElementById('print-controls').style.display = 'block'; } catch (err) { console.error("שגיאה בביצוע הזום:", err); } // 3. כעת מנסים לצייר את הנקודות באובייקט נקי ככל האפשר try { const geometryParams = { wkts: wkts, names: names, geometryType: 0, // 0 מייצג Point ב-GovMap defaultSymbol: { url: 'https://maps.google.com/mapfiles/ms/icons/red-dot.png', // תמונה מאובטחת width: 24, height: 24 }, data: { tooltips: tooltips } }; govmap.displayGeometries(geometryParams).then(function(response) { console.log("הנקודות צוירו בהצלחה:", response); }).catch(function(err) { console.error("שגיאה בציור של GovMap:", err); }); } catch (err) { console.error("שגיאה בהכנת הנתונים לציור:", err); } }, 500); } else { document.getElementById('loading').innerText = "לא נמצאו נקודות עם קואורדינטות בתוך שטח ישראל."; } }
